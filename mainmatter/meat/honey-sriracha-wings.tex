% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[20]{mins}},
    bakingtime={\unit[50]{mins} \unit[200]{\textcelcius}},
    portion = {\portion{4}},
    source = {Internet somewhere}
]
{Honey sriracha wings}
       
    \introduction{%
        Not very spicy wings but they are really sweet. Have to adjust recipe a bit. 
    }
    
    \ingredients{%
        \textbf{Wings}\\
        \unit[1]{kg} & Chicken wings\\
        2 tbsp & Melted unsalted butter\\
        1 tsp & Garlic\\
        & Salt and pepper\\
        & Cumin and turmeric\\
        & Vegetable oil\\
        \textbf{Sauce}\\
        5 tbsp & Unsalted butter\\
        $\frac{1}{3}$ cup & Honey\\
        $\frac{1}{4}$ cup & Sriracha\\
        1 tbsp & Soy sauce\\
        2 tsp & Lime juice
   }
    
    \preparation{%
        \step Marinate the ingredients for the \textbf{Wings} together and leave for a while (longer the better).
        \step Preheat oven to 200\textcelcius .
        \step Bake the wings until browned and crisp (45 to 55 mins).
        \step Mix the ingredients for \textbf{Sauce} together. Dip the wings in the \textbf{Sauce} and serve. 
    }
       
\end{recipe}
