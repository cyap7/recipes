% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[1]{h}},
    portion = {\portion{5-6}},
    source = {Mum}
]
{Bak kua}
       
    \introduction{%
        Malaysian beef jerky. Tastes really good.
    }
    
    \ingredients{%
        \unit[300]{g} & Lean pork mince\\
        \unit[300]{g} & Poon fei san mince\\
        \unit[135]{g} & Sugar\\
        3 tbsp & Fish sauce\\
        1 tsp & Soy sauce\\
        $\frac{1}{4}$ tsp & Salt\\
        $\frac{1}{4}$ tsp & Pepper\\
        1 tsp & Wine
    }
    
    \preparation{%
        \step Mix everything together until gooey then roll out into a thin strip.
        \step Leave in the fridge overnight.
        \step Dry outside under the sun until dry. Takes a couple of hours in the Australian summer.
        \step Cut it up into \unit[10]{cm} squares then fry.
    }
       
    \hint{%
        Get fly swatters ready for the inevitable onslaught. 
    }
    
\end{recipe}
