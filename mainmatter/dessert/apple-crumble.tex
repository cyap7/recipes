% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[20]{mins}},
    bakingtime={\unit[25]{mins} 180\textcelcius},
    portion = {\portion{4}},
    source = {Mum}
]
{Fruit crumble}
   
    \introduction{%
        Apple crumble is super nice in winter when it is cold. 
    }
    
    \ingredients{%
        \unit[425]{g} & Drained canned fruit (or stewed apples)\\
        $\frac{1}{2}$ cup & Plain flour\\
        $\frac{1}{2}$ cup & Rolled oats\\
        $\frac{1}{4}$ cup & Brown sugar\\
        $\frac{1}{2}$ tsp & Cinnamon (optional)\\
        3 tbsp & Melted butter
    }
    
    \preparation{%
        \step Preheat oven to 180\textcelcius.
        \step Mix dry ingredients together.
        \step Add melted butter to the dry mixter and mix well.
        \step Spoon fruit into a \unit[1]{L} ovenproof dish then top with the crumble mixter.
        \step Bake 25-30 mins.
    }
       
    \hint{%
        Mum doubles the recipe for ``8`` people. 
    }
    
\end{recipe}
