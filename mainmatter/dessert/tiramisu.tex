% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[30]{mins}},
    source = {Mum}
]
{Tiramisu}
   
    \introduction{%
        Insert something good.
    }
    
    \ingredients{%
        \unit[250]{g} & Mascarpone\\
        1 tsp & Vanilla\\
        3 tbsp & Tia maria/marsala\\
        \unit[150]{mL} & Strong coffee cooled to room temperature\\
        \unit[120]{mL} & Cream\\
        4 tbsp & Icing sugar\\
        16 & Savioardi biscuits
    }
    
    \preparation{%
        \step Whisk mascarpone + vanilla + 50 mL coffee.
        \step Whisk cream and icing sugar until smooth, then fold into mascarpone.
        \step Pour remaining coffee into a bowl. Dip biscuits so they absorb the coffee then line the tin with biscuits.
        \step Layer cream mixture, then biscuits, then cream mixture.
    }
       
    \hint{%
        Multiply the recipe as necessary.
    }
    
\end{recipe}
