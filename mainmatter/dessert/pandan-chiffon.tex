% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[40]{mins}},
    bakingtime={\unit[40-45]{mins} 180\textcelcius},
    source = {Mum}
]
    {Pandan chiffon cake}
       
    \introduction{%
        Donut shaped extremely fluffy cake. Takes lots of skill to make.
    }
    
    \ingredients{%
        \textbf{A}\\
        4 & Egg yolks\\
        \unit[70]{g} & Caster sugar\\
        $\frac{1}{4}$ tsp & Salt\\
        \textbf{B}\\
        \unit[85]{mL} & Corn oil ($\frac{1}{3}$ cup)\\
        \unit[115]{mL} & Pandan juice/essence\\
        \textbf{C}\\
        \unit[150]{g} & Cake flour\\
        1 tsp & Baking powder\\
        \textbf{D}\\
        4 & Egg whites\\
        \unit[70]{g} & Caster sugar\\
        $\frac{1}{2}$ tsp & Tartar
    }
    
    \preparation{%
        \step Heat oven 170-180\textcelcius.
        \step Sift \textbf{C} twice then set aside.
        \step Cream \textbf{A} until sugar is dissolved. Add \textbf{B} in the order listed. Mix well after each addition.
        \step Add sifted flour and mix well.
        \step Beat egg whites until frothy, add tartar then beat until soft peaks. Add sugar then beat until stiff peaks.
        \step Fold $\frac{1}{2}$ egg white mixter into yolk mixture.
        \step Pour yolk mixture into white mixture gently. 
        \step Pour into pan (donut shaped) and bang against the bench once to release bubbles. Bake for 40-45 mins.
    }
       
    \hint{%
        Beating eggs is the most important part.
    }
    
\end{recipe}
